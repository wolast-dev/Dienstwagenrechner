<script>


function showPage(id, el){{


    document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));


    document.getElementById(id).classList.add('active');


    document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));


    el.classList.add('active');


}}


function berechnen(){{


    let lp = parseFloat(document.getElementById('listenpreis').value)||0;


    let leasing = parseFloat(document.getElementById('leasingrate').value)||0;


    let faktor = parseFloat(document.getElementById('antrieb').value)||0;


    let eigen = parseFloat(document.getElementById('eigen').value)||0;


    let gsv = (lp * faktor) - eigen;


    if(gsv < 0) gsv = 0;


    let steuer = gsv * 0.35;


    let lf = (leasing / lp) * 100;


    document.getElementById('gsv').innerText = gsv.toFixed(2) + " €";


    document.getElementById('steuer').innerText = steuer.toFixed(2) + " €";


    document.getElementById('lf').innerText = lf.toFixed(2) + "%";


    let text = "";


    let klasse = "";


    if(lf < 0.8){{


        text = "Sehr gut";


        klasse = "green";


    }}


    else if(lf < 1.1){{


        text = "Gut";


        klasse = "green";


    }}


    else {{


        text = "Eher teuer";


        klasse = "red";


    }}


    document.getElementById('bewertung').innerHTML =


        '<span class="' + klasse + '">' + text + '</span>';


}}


function speichern(){{


    let fahrzeug = document.getElementById('fahrzeug').value;


    let daten = JSON.parse(localStorage.getItem('autos')) || [];


    daten.push(fahrzeug);


    localStorage.setItem('autos', JSON.stringify(daten));


    laden();


}}


function laden(){{


    let daten = JSON.parse(localStorage.getItem('autos')) || [];

    }});


    if(html === "") html = "Keine Fahrzeuge gespeichert.";


    document.getElementById('liste').innerHTML = html;


}}


berechnen();
